## 🎯 核心说明

**问题：** 如何将数字证书集成到打包流程中？
**答案：** 使用 `build_with_signing.py` 替代原始打包脚本

## 💻 代码示例

### 签名函数调用

```python
# build_with_signing.py (第 279-323 行)

def initialize_signer():
    """初始化签名工具"""
    from code_signer import CodeSigner

    # 加载配置
    config_path = '../signature_config.json'
    signer = CodeSigner.from_config(config_path)

    return signer, 'code_signer'


def sign_exe_file_with_module(exe_path, signer, signer_type):
    """使用签名模块对EXE文件进行数字签名"""

    # 调用签名函数
    success, message = signer.sign_file(
        exe_path,  # EXE 文件路径
        'sap_hour_operation'  # 证书名称
    )

    if success:
        print(f"✓ 签名成功!")
        return True, message
    else:
        print(f"✗ 签名失败: {message}")
        return False, message
```

### code_signer 核心函数

```python
# code_signer/core.py (简化版)

class CodeSigner:
    def sign_file(self, file_path, certificate_name):
        """对文件进行签名"""

        # 1. 获取证书配置
        cert_config = self.config.get_certificate(certificate_name)

        # 2. 查找 signtool.exe
        signtool_path = self.find_signtool()

        # 3. 构建签名命令
        cmd = [
            signtool_path,
            'sign',
            '/sha1', cert_config.sha1,    # 证书 SHA1
            '/fd', 'SHA256',               # 哈希算法
            '/tr', self.timestamp_server,  # 时间戳服务器
            '/td', 'SHA256',
            file_path
        ]

        # 4. 执行签名
        result = subprocess.run(cmd)

        # 5. 返回结果
        if result.returncode == 0:
            return True, "文件签名成功"
        else:
            return False, "签名失败"
```
